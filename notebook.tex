    \subsection{循环之
break，continue，pass}\label{ux5faaux73afux4e4b-breakcontinuepass}

\begin{itemize}
\tightlist
\item
  break：无条件结束整个循环，简称循环猝死
\item
  continue: 无条件结束本次循环，重新进入下一轮循环
\item
  pass: 表示略过，占位符
\end{itemize}
